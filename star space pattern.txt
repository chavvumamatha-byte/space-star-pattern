#include <stdio.h>
int main()
{
   int n,rows,cols,count=1,spaces;
   
   printf("Enter a Number : ");
   scanf("%d" , &n);
   
   for(rows = 1; rows <=n; rows++)
   {
       for(spaces = 1; spaces <=n - rows; spaces++)
       {
           printf(" ");
       }
       for(cols = 1; cols <= n - rows + 1; cols++)
       {
           printf("*  ");
   }
    printf("\n");
   }
   return 0;
}
